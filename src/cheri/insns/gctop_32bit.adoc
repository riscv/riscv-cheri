Description::
Decode the top integer address from `{cs1}` 's bounds, saturated to 2^XLEN^-1, and write the result to `rd`.

let capVal = C(cs1);
X(rd) = match getCapBoundsBits(capVal) {
  None() => zeros(),
  Some(_, top) => {
    let top_val = unsigned(top);
    to_bits(xlen, if top_val > cap_max_addr then cap_max_addr else top_val)
